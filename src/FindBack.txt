Sub FindBack()
' Paul Beverley - Version 24.02.23
' Next find backwards

Set rng = Selection.Range.Duplicate
Selection.Collapse wdCollapseStart
With Selection.Find
  .Wrap = wdFindStop
  .Forward = False
  .Execute
  If .Found = False Then Beep: GoTo myEnd
  If Selection.End < Selection.Start Then
    ' Word's Find is getting confused, so exit
    rng.Select
    Beep
    GoTo myEnd
  End If
  If .Found = True Then
    Selection.Collapse wdCollapseStart
    Selection.MoveLeft , 1
    .Forward = True
    .Execute
  End If
myEnd:
' Leave F&R dialogue in a sensible state
  .Wrap = wdFindContinue
End With
End Sub
