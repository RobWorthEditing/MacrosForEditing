Sub FindBack()
' Paul Beverley - Version 10.03.23
' Next find backwards

' Set rng = Selection.Range.Duplicate
Selection.Collapse wdCollapseStart
With Selection.Find
  .Wrap = wdFindStop
  .Forward = False
  .MatchCase = False
  If InStr(.Text, "]") > 0 Then .MatchWildcards = True
  .Execute
  If .Found = False Then Beep
' Leave F&R dialogue in a sensible state
  .Wrap = wdFindContinue
  .Forward = True
End With

If Selection.Find.Found = False Then
  Beep
Else
  If Selection.Start = rng.Start Then
    rng.Select
    Beep
    myTime = Timer
    Do
    Loop Until Timer > myTime + 0.2
    Beep
    StatusBar = "Sorry, Word's Find facility is playing sillies!"
  End If
End If
End Sub
