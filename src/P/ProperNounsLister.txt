
Sub ProperNounsLister()
' Paul Beverley - Version 28.09.25
' Creates a list of all proper nouns, excluding sentence starts

minLength = 3

allExcepts = ""
For Each myDoc In Documents
  If InStr(myDoc.Content, "ProperNounsLister") > 0 Then
    allExcepts = LCase(myDoc.Content)
    Exit For
  End If
  DoEvents
Next myDoc

Set dict = CreateObject("Scripting.Dictionary")
For Each myWd In ActiveDocument.Content.words
  cleanWord = Trim(myWd)
  ' Skip short words
  myCap = Left(cleanWord, 1)
  xkjdfhkh = LCase(myCap) <> myCap
  jfkjsxfc = (Len(cleanWord) >= minLen)
  If (LCase(myCap) <> myCap) And (Len(cleanWord) >= minLength) Then
    If InStr(allExcepts, vbCr & LCase(cleanWord) & vbCr) = 0 Then
      If Not dict.Exists(cleanWord) Then dict.Add cleanWord, 1
    End If
  End If
Next myWd

Documents.Add
Dim key As Variant
For Each key In dict.keys
  Selection.TypeText key & vbCr
  Debug.Print
Next key
Set rng = ActiveDocument.Content
rng.Sort
Selection.InsertAfter Text:=vbCr
rng.Collapse wdCollapseStart
Do
 rng.MoveEnd , 1
Loop Until rng.Characters.last <> vbCr
rng.MoveEnd , -1
rng.Delete
rng.InsertBefore Text:="Proper nouns (excluding sentence starts):" & vbCr & vbCr
Selection.HomeKey Unit:=wdStory
Beep
End Sub
