
Sub CopyToList()
' Paul Beverley - Version 07.02.26
' Copies selected text into a list file

' keyWord = "queries"
keyWord = "tmp"
keyWord = "list"

' copyWholePara = False
copyWholePara = True

copySentenceInstead = False

addPageLineNumbers = False

includeFormatting = True
' includeFormatting = False

addReturn = True
' addBlankLine = False
addBlankLine = True

alwaysCopyAtEnd = False
' alwaysCopyAtEnd = True

goBackToSource = True
' goBackToSource = False

wordsToAvoid = "switch"
' wordsToAvoid = "FRedit,switch"

myHighlightColour = wdNoHighlight
' To add highlight, use:
' myHighlightColour = wdYellow

addPageLineNumbers = True
Set thisDoc = ActiveDocument
Set rng = Selection.Range.Duplicate
rng.Collapse wdCollapseStart
If addPageLineNumbers = True Then
  totalPages = thisDoc.ComputeStatistics(wdStatisticPages)
  thisPage = rng.Information(wdActiveEndAdjustedPageNumber)
  rng.End = Selection.End
  thisPageEnd = rng.Information(wdActiveEndAdjustedPageNumber)
  Set thisDoc = ActiveDocument
  If addPageLineNumbers = True Then
    totalPages = thisDoc.ComputeStatistics(wdStatisticPages)
    If pbPageCount <> totalPages Then
      myResponse = MsgBox("Page count mismatch. Is the cursor in the right document?", _
         vbQuestion + vbYesNo, "CopyToList")
      If myResponse <> vbYes Then Exit Sub
      ReDim pbLineCount(totalPages) As Long
      myResponse = MsgBox("Recount the lines on each page?", _
         vbQuestion + vbYesNo, "CopyToList")
      If myResponse <> vbYes Then Exit Sub
      
      Set rngNow = Selection.Range.Duplicate
      Selection.HomeKey Unit:=wdStory
      Application.ScreenUpdating = False
      On Error GoTo ReportIt
      myCount = 1
      For myPage = 1 To totalPages - 1
        Do
          Selection.MoveDown , 1
          thisPage = Selection.Range.Information(wdActiveEndAdjustedPageNumber)
          myCount = myCount + 1
        Loop Until thisPage > myPage
        pbLineCount(thisPage) = myCount - 1
        DoEvents
        displayCount = displayCount + 1
        If thisPage Mod 10 = 0 Then
          Application.ScreenUpdating = True
          For i = 1 To 10: DoEvents: Next i
          Application.ScreenUpdating = False
        End If
      Next myPage
      rngNow.Select
      pbPageCount = totalPages
      Application.ScreenUpdating = True
      Exit Sub
    Else
      Set rng = Selection.Range.Duplicate
      myStLine = Selection.Range.Information(wdFirstCharacterLineNumber)
      Selection.Collapse wdCollapseEnd
      Selection.MoveEnd , -1
      myEndLine = Selection.Range.Information(wdFirstCharacterLineNumber)
      myCount = pbLineCount(thisPage) + myStLine
      myEndCount = pbLineCount(thisPageEnd) + myEndLine
      rng.Select
    End If
  End If
End If

wds = Split("," & LCase(wordsToAvoid), ",")

If Selection.Start = Selection.End Then
  If copyWholePara = True Then
    If copySentenceInstead = True Then
      Selection.Expand wdSentence
    Else
      Selection.Expand wdParagraph
    End If
  Else
    Selection.Expand wdWord
    Do While InStr(ChrW(8217) & "' ", Right(Selection.Text, 1)) > 0
      Selection.MoveEnd , -1
      DoEvents
    Loop
  End If
Else
  Set rng = Selection.Range.Duplicate
  rng.Collapse wdCollapseEnd
  rng.MoveEnd , -1
  rng.Expand wdWord
  If LCase(rng) <> UCase(rng) Then
    Do While InStr(ChrW(8217) & "' ", Right(rng.Text, 1)) > 0
      rng.MoveEnd , -1
      DoEvents
    Loop
  End If
  Selection.Collapse wdCollapseStart
  Selection.Expand wdWord
  Selection.Collapse wdCollapseStart
  rng.Start = Selection.Start
  rng.Select
End If
Set sourceText = Selection.Range.Duplicate

gottaList = False
For Each myDoc In Application.Documents
  thisName = myDoc.Name
  nm = LCase(thisName)
  gottaList = False
  If InStr(nm, LCase(keyWord)) > 0 Then gottaList = True
  For i = 1 To UBound(wds)
    If InStr(nm, wds(i)) > 0 Then gottaList = False
  Next i
  If gottaList = True Then Exit For
Next myDoc

If gottaList = False Then
  Beep
  MsgBox "Can't find a list."
  Exit Sub
End If

myDoc.Activate
Set rng = Selection.Range.Duplicate

If alwaysCopyAtEnd = True Then
  rng.Start = myDoc.Content.End
  rng.Collapse wdCollapseEnd
Else
  hereNow = rng.Start
  rng.Expand wdParagraph
  If rng.Start = hereNow Then rng.Collapse wdCollapseStart
  If Len(rng) = 1 Then
    rng.Collapse wdCollapseStart
  Else
    rng.Collapse wdCollapseEnd
  End If
End If

If addPageLineNumbers = True Then
  myText = vbCr & "(page " & Trim(Str(thisPage)) & _
  ", lines " & Trim(Str(myCount)) & _
      ChrW(8211) & Trim(Str(myEndCount)) & ")"
  rng.InsertBefore Text:=myText & vbCr
  rng.Collapse wdCollapseEnd
End If

If includeFormatting = True Then
  rng.FormattedText = sourceText.FormattedText
Else
  rng.Text = sourceText.Text
End If

rng.Start = rng.Start + Len(sourceText)
rng.Select

If InStr(sourceText, vbCr) = 0 Then
  rng.InsertBefore Text:=vbCr
End If

If addBlankLine = True Then
  rng.InsertBefore Text:=vbCr
End If

thisDoc.Activate
If myHighlightColour > 0 Then
  sourceText.HighlightColorIndex = myHighlightColour
  Selection.Collapse wdCollapseEnd
End If
If goBackToSource = False Then myDoc.Activate
Exit Sub


' Switch the screen back on if there's an error
ReportIt:
Application.ScreenUpdating = True
On Error GoTo 0
Resume
End Sub
