
Sub CommentDelete()
' Paul Beverley - Version 15.12.25
' Deletes current comment

Set rng = Selection.Range.Duplicate
rng.start = 0
n = rng.Comments.Count
If Selection.Information(wdInCommentPane) = False Then   n = n + 1

ActiveDocument.Comments(n).Scope.Select
ActiveDocument.Comments(n).DeleteRecursively
Selection.Collapse wdCollapseEnd
End Sub
