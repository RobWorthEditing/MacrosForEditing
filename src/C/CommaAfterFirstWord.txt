
Sub CommaAfterFirstWord()
' Paul Beverley - Version 12.12.24
' Adds a comma after the first word of the current paragraph

doMoveDown = True

Set pa = Selection.Range.Duplicate
pa.Expand wdParagraph
Set rng = pa.Words(1)
rng.Collapse wdCollapseEnd
rng.Select
Selection.MoveEnd , -1
Selection.InsertAfter Text:=","
If doMoveDown = True Then Selection.MoveDown , 1
End Sub
