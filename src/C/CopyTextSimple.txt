
Sub CopyTextSimple()
' Paul Beverley - Version 11.11.25
' Creates a text-only copy, with some features preserved

doWait = False
' If Word crashes, try again, but with
' doWait = True

doItalic = True
doBold = True
doSub = True
doSuper = True
doListNumbers = True

myColour = wdYellow

ActiveDocument.TrackRevisions = False
oldColour = Options.DefaultHighlightColorIndex
Options.DefaultHighlightColorIndex = myColour
Set rngOld = ActiveDocument.Content
myLanguage = rngOld.Characters(1).LanguageID
Set newDoc = Documents.Add
newDoc.Content.FormattedText = rngOld.FormattedText

myScreenOff = True
' myScreenOff = False

If myScreenOff = True Then
  Application.ScreenUpdating = False
  On Error GoTo ReportIt
End If
Application.ScreenUpdating = False


If doListNumbers = True Then newDoc.ConvertNumbersToText
Set rng = newDoc.Content
rng.Revisions.AcceptAll
rng.Fields.Unlink
rng.LanguageID = myLanguage
If numberCmnts > 0 Then newDoc.DeleteAllComments

wds1 = rng.Words.Count
rng.Font.Hidden = False
wds2 = rng.Words.Count
If wds2 <> wds1 Then
  WordBasic.EditUndo
  myResponse = MsgBox("Original contains " & wds2 - wds1 & _
       " words of hidden text. Include it?", _
       vbQuestion + vbYesNoCancel, "CopyText")
  If myResponse = vbCancel Then Exit Sub
  If myResponse = vbYes Then
    newDoc.ActiveWindow.View.ShowHiddenText = True
    With Selection.Find
      .ClearFormatting
      .Replacement.ClearFormatting
      .Text = ""
      .Font.Hidden = True
      .Wrap = wdFindContinue
      .Replacement.Font.Hidden = False
      .Replacement.Highlight = True
      .Replacement.Text = ""
      .Execute Replace:=wdReplaceAll
      DoEvents
    End With
    newDoc.ActiveWindow.View.ShowHiddenText = False
  End If
End If

Selection.HomeKey Unit:=wdStory
myComments = ""
numNotes = newDoc.Endnotes.Count
If numNotes > 0 Then
  myComments = myComments & "| endnotes = yes (" _
       & Trim(Str(numNotes)) & ")" & vbCr
  Set rng = newDoc.Content
  rng.Collapse wdCollapseEnd
  rng.InsertAfter Text:=vbCr & "Endnotes:" & vbCr & vbCr
  rng.Collapse wdCollapseEnd
  rng.Text = _
       newDoc.StoryRanges(wdEndnotesStory).Text
  ' Delete all notes
  For j = numNotes To 1 Step -1
    newDoc.Endnotes(j).Delete
    DoEvents
  Next j
End If

numNotes = newDoc.Footnotes.Count
If numNotes > 0 Then
  myComments = myComments & "| footnotes = yes (" _
       & Trim(Str(numNotes)) & ")" & vbCr
  Set rng = newDoc.Content
  rng.Collapse wdCollapseEnd
  rng.InsertAfter Text:=vbCr & "Footnotes:" & vbCr & vbCr
  rng.Collapse wdCollapseEnd
  rng.Text = _
       newDoc.StoryRanges(wdFootnotesStory).Text
  ' Delete all notes
  For j = numNotes To 1 Step -1
    newDoc.Footnotes(j).Delete
    DoEvents
  Next j
End If

' copy all the textboxes to the end of the text
shCount = newDoc.Shapes.Count
tbCount = 0
If shCount > 0 Then
  Selection.EndKey Unit:=wdStory
  Selection.TypeText vbCr & "Textboxes:" & vbCr & vbCr
  Set rngTxt = Selection.Range.Duplicate
  rngTxt.Collapse wdCollapseEnd
  For j = 1 To shCount
    Set shp = newDoc.Shapes(j)
    If shp.Type <> 24 And shp.Type <> 3 Then
      If shp.TextFrame.HasText Then
        Set rng = shp.TextFrame.TextRange
        If Len(rng.Text) > 1 Then
          tbCount = tbCount + 1
          rngTxt.Text = rng.Text
          rngTxt.Collapse wdCollapseEnd
        End If
      End If
    End If
    DoEvents
  Next
  If tbCount > 0 Then myComments = myComments & _
       "| textboxes = yes (" & Trim(Str(tbCount)) _
       & ")" & vbCr
End If

' Add a newline for safety
rngText = vbCr
Selection.HomeKey Unit:=wdStory
If myComments > "" Then Selection.TypeText myComments & vbCr

Application.ScreenUpdating = True
StatusBar = "                                                                       Number 1"
For i = 1 To 100
  DoEvents
Next i
Application.ScreenUpdating = False

' Watch out for hard spaces and spaced dots for ellipses
Set rng = ActiveDocument.Content
With rng.Find
  .ClearFormatting
  .Replacement.ClearFormatting
  .Text = "^s"
  .Wrap = wdFindContinue
  .Replacement.Text = " "
  .MatchWildcards = False
  .Execute Replace:=wdReplaceAll
  DoEvents
  
  .ClearFormatting
  .Replacement.ClearFormatting
  .Text = ". . ."
  .Wrap = wdFindContinue
  .Replacement.Text = "…"
  .Execute Replace:=wdReplaceAll
  DoEvents
  
  .ClearFormatting
  .Replacement.ClearFormatting
  .Text = ""
  .Highlight = True
  .Wrap = wdFindContinue
  .Replacement.Text = "vkvk^&kvkv"
  .MatchCase = True
  .MatchWildcards = False
  .Execute Replace:=wdReplaceAll
  DoEvents

  If doItalic = True Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = ""
    .Font.Italic = True
    .Wrap = wdFindContinue
    .Replacement.Text = "zczc^&czcz"
    .MatchCase = True
    .MatchWildcards = False
    .Execute Replace:=wdReplaceAll
    DoEvents
  End If

  If doBold = True Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = ""
    .Font.Bold = True
    .Wrap = wdFindContinue
    .Replacement.Text = "jqjq^&qjqj"
    .MatchCase = True
    .MatchWildcards = False
    .Execute Replace:=wdReplaceAll
    DoEvents
  End If

  If doSub = True Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = ""
    .Font.Subscript = True
    .Wrap = wdFindContinue
    .Replacement.Text = "xbxb^&bxbx"
    .MatchCase = True
    .MatchWildcards = False
    .Execute Replace:=wdReplaceAll
    DoEvents
  End If

  If doSuper = True Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = ""
    .Font.Superscript = True
    .Wrap = wdFindContinue
    .MatchCase = True
    .Replacement.Text = "xsxs^&sxsx"
    .MatchWildcards = False
    .Execute Replace:=wdReplaceAll
  End If
End With

Application.ScreenUpdating = True
StatusBar = "                                                                       Number 2"
For i = 1 To 100
  DoEvents
Next i
Application.ScreenUpdating = False

' Copy pure text after the end of the formatted text
If doWait = True Then MsgBox "Difficult step"
allText = newDoc.Content.Text
sdasd = Len(allText)
allText = Replace(allText, Asc(12), vbCr)
sasd = Len(allText)

If doWait = True Then MsgBox "Next step"
newDoc.Close SaveChanges:=False
If doWait = True Then MsgBox "Next step"
Set newDoc2 = Documents.Add
If doWait = True Then MsgBox "Next step"
Selection.TypeText Text:=allText

Application.ScreenUpdating = True
StatusBar = "                                                                       Number 3"
For i = 1 To 100
  DoEvents
Next i
Application.ScreenUpdating = False

If doWait = True Then MsgBox "One more step"
Set rng = newDoc2.Content
If doWait = False Then On Error Resume Next

With rng.Find
 ' .ClearFormatting
 ' .Replacement.ClearFormatting
 ' .Text = "^12"
 ' .Wrap = wdFindContinue
 ' .Replacement.Text = "vbnvbn"
 ' .MatchWildcards = False
 ' .Execute
 ' If .Found = True Then .Execute Replace:=wdReplaceAll
  
 ' .Text = "^14"
 ' .Execute
 ' If .Found = True Then .Execute Replace:=wdReplaceAll
  
If doWait = True Then MsgBox ("Count to 20")

  If doSuper = True Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "xsxs(*)sxsx"
    .Wrap = wdFindContinue
    .Replacement.Text = "\1"
    .Replacement.Font.Superscript = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll
    
If doWait = True Then MsgBox ("1")
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "XSXS(*)SXSX"
    .Wrap = wdFindContinue
    .Replacement.Text = "\1"
    .Replacement.Font.Superscript = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("2")
  End If

  If doSub = True Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "xbxb(*)bxbx"
    .Wrap = wdFindContinue
    .Replacement.Text = "\1"
    .Replacement.Font.Subscript = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll

If doWait = True Then MsgBox ("3")
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "XBXB(*)BXBX"
    .Wrap = wdFindContinue
    .Replacement.Text = "\1"
    .Replacement.Font.Subscript = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("4")
  End If

  If doBold = True Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "jqjq(*)qjqj"
    .Wrap = wdFindContinue
    .Replacement.Text = "\1"
    .Replacement.Font.Bold = True
    .MatchCase = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("5")
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "JQJQ(*)QJQJ"
    .Wrap = wdFindContinue
    .Replacement.Text = "\1"
    .Replacement.Font.Bold = True
    .MatchCase = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("6")
  End If

  If doItalic = True Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "zczc(*)czcz"
    .Wrap = wdFindContinue
    .Replacement.Text = "\1"
    .Replacement.Font.Italic = True
    .MatchCase = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("7")
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "ZCZC(*)CZCZ"
    .Wrap = wdFindContinue
    .Replacement.Text = "\1"
    .Replacement.Font.Italic = True
    .MatchCase = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("8")
  End If

' Restore Highlight
  .ClearFormatting
  .Replacement.ClearFormatting
  .Text = "vkvk(*)kvkv"
  .Format = True
  .Wrap = wdFindContinue
  .Replacement.Text = "\1"
  .Replacement.Highlight = True
  .MatchCase = False
  .MatchWildcards = True
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("9")

Options.DefaultHighlightColorIndex = oldColour
' Tidy up picture markers
  .ClearFormatting
  .Replacement.ClearFormatting
  .Text = "^1"
  .Wrap = wdFindContinue
  .Replacement.Text = ""
  .MatchWildcards = False
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("10")

  .ClearFormatting
  .Replacement.ClearFormatting
  .Text = "vkvk"
  .Wrap = wdFindContinue
  .Replacement.Text = ""
  .MatchCase = False
  .MatchWildcards = False
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("11")

  .Text = "kvkv"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("12")

  .Text = "xbxb"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll

If doWait = True Then MsgBox ("13")
  .Text = "bxbx"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("14")

  .Text = "xsxs"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("15")

  .Text = "sxsx"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("16")

  .Text = "jqjq"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("17")

  .Text = "qjqj"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll
If doWait = True Then MsgBox ("18")

  .Text = "zczc"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll

If doWait = True Then MsgBox ("19")
  .Text = "czcz"
  .Execute
  If .Found = True Then .Execute Replace:=wdReplaceAll

  If tbCount = 0 Then
    .ClearFormatting
    .Replacement.ClearFormatting
    .Text = "Textboxes:"
    .Wrap = wdFindContinue
    .Replacement.Text = ""
    .MatchCase = True
    .MatchWildcards = True
    .Execute
    If .Found = True Then .Execute Replace:=wdReplaceAll
  End If
End With
If doWait = True Then MsgBox ("20")
Application.ScreenUpdating = True
Selection.HomeKey Unit:=wdStory
Beep
Exit Sub

' Switch the screen back on if there's an error
ReportIt:
Application.ScreenUpdating = True
On Error GoTo 0
Resume
End Sub
