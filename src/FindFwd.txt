Sub FindFwd()
' Paul Beverley - Version 10.03.23
' Finds next match forwards, case insensitively

Set rng = Selection.Range.Duplicate
Selection.Collapse wdCollapseEnd
With Selection.Find
  .Wrap = wdFindStop
  .Forward = True
  .MatchCase = False
  .Execute
  If InStr(.Text, "]") > 0 Then .MatchWildcards = True
' Leave F&R dialogue in a sensible state
  .Wrap = wdFindContinue
End With
If Selection.Find.Found = False Then
  Beep
Else
  If Selection.End = 0 Then
    rng.Select
    Beep
    myTime = Timer
    Do
    Loop Until Timer > myTime + 0.2
    Beep
    Selection.EndKey Unit:=wdStory
    With Selection.Find
      .Wrap = wdFindStop
      .Forward = False
      .Execute
    ' Leave F&R dialogue in a sensible state
      .Wrap = wdFindContinue
      .Forward = True
    End With
    StatusBar = "Sorry, Word's Find facility is playing sillies!"
  End If
End If
End Sub
