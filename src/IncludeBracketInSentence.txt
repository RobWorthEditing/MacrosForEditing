Sub IncludeBracketInSentence()
' Paul Beverley - Version 24.08.12
' Move the full point to after the bracketed reference

myTrack = ActiveDocument.TrackRevisions
ActiveDocument.TrackRevisions = False
If Selection.Start = Selection.End Then Selection.MoveEnd wdCharacter, 40

Selection.MoveEndWhile cset:=ChrW(13), Count:=wdBackward
mm = Selection
mm = Replace(mm, ".", "")
mm = Replace(mm, ")", ").")
mm = Replace(mm, ")..", ").")
Selection.TypeText mm

ActiveDocument.TrackRevisions = myTrack
End Sub
